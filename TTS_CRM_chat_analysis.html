지난달 제품별 판매 비중 차트로 보여줘
SQL문 보기 
SELECT
  p.product_name,
  SUM(s.quantity) AS total_sales_count,
  ROUND(SUM(s.quantity) * 100.0
    / SUM(SUM(s.quantity)) OVER(), 2
  ) AS sales_ratio_percent
FROM sales s
JOIN products p
  ON s.product_id = p.product_id
